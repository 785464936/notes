```dataviewjs
const pages = dv.pages()
  .where(p =>
    p.project === "学英语" &&
    !p.file.path.includes("Templates") &&
    !(p.tags && p.tags.includes("english-word")) &&
    !(p.tags && p.tags.includes("phrases"))
  )
  .sort(p => p.file.name, 'asc');

// =====================
// 1. 标题区（总览）
// =====================
dv.header(2, "📘 英语学习面板");
dv.paragraph(`共 ${pages.length} 条记录`);

// =====================
// 2. 按类型分组（如果你有 tags）
// =====================
const grouped = {
  "📄 普通内容": [],
  "🧠 单词": [],
  "💬 句型/短语": []
};

for (let p of pages) {
  if (p.tags?.includes("english-word")) grouped["🧠 单词"].push(p);
  else if (p.tags?.includes("phrases")) grouped["💬 句型/短语"].push(p);
  else grouped["📄 普通内容"].push(p);
}

// =====================
// 3. 渲染函数（统一样式）
// =====================
function renderGroup(title, items) {
  dv.header(3, `${title} (${items.length})`);

  if (items.length === 0) {
    dv.paragraph("暂无内容");
    return;
  }

  dv.table(
    ["标题", "路径"],
    items.map(p => [p.file.link, p.file.path])
  );
}

// =====================
// 4. 输出各组
// =====================
renderGroup("🧠 单词", grouped["🧠 单词"]);
renderGroup("💬 句型/短语", grouped["💬 句型/短语"]);
renderGroup("📄 普通内容", grouped["📄 普通内容"]);
```
